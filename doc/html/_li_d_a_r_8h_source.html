145 ID2 = __ID2;

147 color = __color;
148
149 //calculate area
150 helios::vec3 s0 = vertex1-vertex0;
151 helios::vec3 s1 = vertex2-vertex0;
152 helios::vec3 s2 = vertex2-vertex1;
153
154 float a = s0.magnitude();
155 float b = s1.magnitude();
156 float c = s2.magnitude();

159 area = sqrt( s*(s-a)*(s-b)*(s-c) );
