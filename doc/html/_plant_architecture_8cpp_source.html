21helios::vec3 interpolateTube( const std::vector<vec3> &P, float frac ){
22
23 assert( frac>=0 && frac<=1 );
24 assert( !P.empty() );
25
26 float dl=0.f;
27 for( int i=0; i<P.size()-1; i++ ){
28 dl+=(P.at(i+1)-P.at(i)).magnitude();
29 }
30
31 float f = 0;
32 for( int i=0; i<P.size()-1; i++ ){
33
34 float dseg = (P.at(i+1)-P.at(i)).magnitude();
35
36 float fplus = f+dseg/dl;
37
38 if( fplus>=1.f ){
39 fplus = 1.f+1e-3;
40 }
41
42 if( frac>=f && (frac<=fplus || fabs(frac-fplus)<0.0001 ) ){
43
44 vec3 V = P.at(i) + (frac-f)/(fplus-f)*(P.at(i+1)-P.at(i));
45
46 return V;
47 }
48
49 f=fplus;
50 }
51
52 return P.front();
53
54}

101helios::vec3 Phytomer::getAxisVector( float stem_fraction, const std::vector<helios::vec3> &axis_vertices ) const{
102
103 if( stem_fraction<0 || stem_fraction>1 ){
104 std::cerr << "ERROR: Stem fraction must be between 0 and 1." << std::endl;
105 }
106
107 float df = 0.1f;
108 float frac_plus, frac_minus;
109 if( stem_fraction+df<=1 ) {
110 frac_minus = stem_fraction;
111 frac_plus = stem_fraction + df;
112 }else{
113 frac_minus = stem_fraction-df;
114 frac_plus = stem_fraction;
115 }
116
117 vec3 node_minus = interpolateTube( axis_vertices, frac_minus );
118 vec3 node_plus = interpolateTube( axis_vertices, frac_plus );
119
120 vec3 norm = node_plus-node_minus;
121 norm.normalize();
122
123 return norm;
