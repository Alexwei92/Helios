21helios::vec3 interpolateTube( const std::vector<vec3> &P, float frac ){
22
23 assert( frac>=0 && frac<=1 );
24 assert( !P.empty() );
25
26 float dl=0.f;
27 for( int i=0; i<P.size()-1; i++ ){
28 dl+=(P.at(i+1)-P.at(i)).magnitude();
29 }
30
31 float f = 0;
32 for( int i=0; i<P.size()-1; i++ ){
33
34 float dseg = (P.at(i+1)-P.at(i)).magnitude();
35
36 float fplus = f+dseg/dl;
37
38 if( fplus>=1.f ){
39 fplus = 1.f+1e-3;
40 }
41
42 if( frac>=f && (frac<=fplus || fabs(frac-fplus)<0.0001 ) ){
43
44 vec3 V = P.at(i) + (frac-f)/(fplus-f)*(P.at(i+1)-P.at(i));
45
46 return V;
47 }
48
49 f=fplus;
50 }
51
52 return P.front();
53
54}

191helios::vec3 Phytomer::getInternodeAxisVector(float stem_fraction) const{
192
193 if( stem_fraction<0 || stem_fraction>1 ){
194 std::cerr << "ERROR: Stem fraction must be between 0 and 1." << std::endl;
195 }
196
197 float df = 0.1f;
198 float frac_plus, frac_minus;
199 if( stem_fraction+df<=1 ) {
200 frac_minus = stem_fraction;
201 frac_plus = stem_fraction + df;
202 }else{
203 frac_minus = stem_fraction-df;
204 frac_plus = stem_fraction;
205 }
206
207 vec3 node_minus = interpolateTube( internode_vertices, frac_minus );
208 vec3 node_plus = interpolateTube( internode_vertices, frac_plus );
209
210 vec3 norm = node_plus-node_minus;
211 norm.normalize();
212
213 return norm;
