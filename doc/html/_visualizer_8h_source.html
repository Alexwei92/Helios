174 void set(const std::vector<helios::RGBcolor> &ctable, const std::vector<float> &clocs, int size, float minval_, float maxval_ ){
175
176 cmapsize = size;
177 minval = minval_;
178 maxval = maxval_;
179
180 size_t Ncolors=ctable.size();
181
182 assert( clocs.size() == Ncolors && minval_<maxval_ );
183
184 cmap.resize(Ncolors);
185
186 float cmin,cmax;
187 std::vector<float> cinds;
188 cinds.resize(Ncolors);
189
190 for( int i=0; i<Ncolors; i++ ){
191 cinds.at(i)=clocs.at(i)*float(cmapsize-1);
192 }
193
194 cmap.resize(cmapsize);
195 for( int c=0; c<Ncolors-1; c++ ){
196
197 cmin=cinds.at(c);
198 cmax=cinds.at(c+1);
199
200 for( int i=0; i<cmapsize; i++){
201
202 if( float(i)>=cmin && float(i)<=cmax ){
203
204 cmap.at(i).r=ctable.at(c).r+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).r-ctable.at(c).r);
205 cmap.at(i).g=ctable.at(c).g+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).g-ctable.at(c).g);
206 cmap.at(i).b=ctable.at(c).b+(float(i)-cmin)/(cmax-cmin)*(ctable.at(c+1).b-ctable.at(c).b);
207
208 }
209 }
210 }
211
212 }

214 helios::RGBcolor query( float x ) const{
215
216 assert( cmapsize>0 && !cmap.empty() );
217
218 helios::RGBcolor color;
219
220 int color_ind;
221 if( minval==maxval ){
222 color_ind=0;
223 }else{
224 color_ind=std::round( (x-minval)/(maxval-minval) * float(cmapsize-1) );
225 }
226
227 if(color_ind<0){color_ind=0;}
228 if(color_ind>cmapsize-1){color_ind=int(cmapsize-1);}
229 color.r=cmap.at(color_ind).r;
230 color.g=cmap.at(color_ind).g;
231 color.b=cmap.at(color_ind).b;
232
233 return color;
