999 helios::vec4 AerialLiDARcloud::RANSAC( const int maxIter, const float threshDist, const float inlierRatio, const std::vector<helios::vec3>& hits, std::vector<bool>& inliers ){
1000 
1001  int N = hits.size();
1002 
1003  //trying to find a model in the form of Ax+By+Cz+D=0
1004  //bestModel.x = A, bestModel.y = B, bestModel.z = C, bestModel.w = D
1005  helios::vec4 bestModel;
1006  float bestError = 1e20;
1007 
1008  for( int iter=0; iter<maxIter; iter++ ){
1009 
1010  helios::vec4 maybeModel;
1011 
1012  //generate 3 random points
1013  int i0 = randu(0,N-1);
1014  int i1 = randu(0,N-1);
1015  int i2 = randu(0,N-1);
1016 
1017  //get x,y,z coordinates for these points
1018  helios::vec3 x0 = hits.at(i0);
1019  helios::vec3 x1 = hits.at(i1);
1020  helios::vec3 x2 = hits.at(i2);
1021 
1022  //determine 'maybeModel' for these three points
1023  helios::vec3 n = cross( x2-x0, x1-x0 );
1024  maybeModel.x = n.x;
1025  maybeModel.y = n.y;
1026  maybeModel.z = n.z;
1027 
1028  maybeModel.w = -(n*x2);
1029 
1030  //loop over all other points
1031  float errsum = 0.f;
1032  int Ninlier = 0;
1033  for( int p=0; p<N; p++ ){
1034 
1035  if( p!=i0 && p!=i1 && p!=i2 ){
1036 
1037  helios::vec3 point = hits.at(p);
1038 
1039  float zplane = -(maybeModel.x*point.x+maybeModel.y*point.y+maybeModel.w)/maybeModel.z;
1040  float zerr = fabs( zplane - point.z ); //NOTE: since we know the plane will be somewhat close to horizontal, we're only using the vertical distance to calculate the error of fit.
1041 
1042  if( zerr<=threshDist ){
1043  errsum += zerr;
1044  Ninlier += 1;
1045  }
1046 
1047  }
1048 
1049  }
1050 
1051  //check if this model is better than the current 'best' model
1052  if( errsum/float(Ninlier)<bestError && float(Ninlier)/float(N)>inlierRatio ){
1053 
1054  //update the 'best' model
1055  bestModel = maybeModel;
1056  bestError = errsum/float(Ninlier);
1057 
1058  }
1059 
1060  }//end iterations loop (iter)
1061 
1062  //separate inliers and outliers based on 'best' model
1063  inliers.resize(N,0);
1064  for( int p=0; p<N; p++ ){
1065 
1066  helios::vec3 point = hits.at(p);
1067 
1068  float zplane = -(bestModel.x*point.x+bestModel.y*point.y+bestModel.w)/bestModel.z;
1069  float zerr = fabs( zplane - point.z );
1070 
1071  if( zerr<=threshDist ){
1072  inliers.at(p) = true;
1073  }
1074 
1075  }
1076 
1077  return bestModel;
1078 
1079 }
