472helios::vec3 VoxelIntersection::linesIntersection(helios::vec3 line1_point, helios::vec3 line1_direction, helios::vec3 line2_point, helios::vec3 line2_direction){
473
474 helios::vec3 g = line2_point-line1_point;
475 helios::vec3 h = cross(line2_direction,g);
476 helios::vec3 k = cross(line2_direction,line1_direction);
477 helios::vec3 rht;
478 helios::vec3 out;
479
480 float h_mag = sqrt(pow(h.x,2) + pow(h.y,2) + pow(h.z,2) );
481 float k_mag = sqrt(pow(k.x,2) + pow(k.y,2) + pow(k.z,2) );
482
483 //in the same direction
484 if( ((h.x >= 0 && k.x >=0) || (h.x < 0 && k.x < 0)) && ((h.y >= 0 && k.y >=0) || (h.y < 0 && k.y < 0)) && ((h.z >= 0 && k.z >=0) || (h.z < 0 && k.z < 0)) ){
485 rht = (h_mag/k_mag)*line1_direction;
486 return line1_point + rht;
487 }else{ // different direction
488 rht = (h_mag/k_mag)*line1_direction;
489 return line1_point - rht ;
