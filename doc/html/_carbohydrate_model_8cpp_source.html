52float Phytomer::calculatePhytomerConstructionCosts(){
53
54 //\todo make these values externally settable
55 float leaf_construction_cost_base = 5.57; //mol C/m^2
56 float internode_construction_cost_base = 39028; //mol C/m^3
57
58 float phytomer_carbon_cost = 0.f; //mol C
59
60 //leaves (cost per area basis)
61 float leaf_area = 0;
62 for( const auto &petiole : leaf_objIDs ) {
63 for( uint leaf_objID : petiole ) {
64 leaf_area += context_ptr->getObjectArea(leaf_objID);
65 }
66 }
67 leaf_area /= current_leaf_scale_factor; //convert to fully-expanded area
68 phytomer_carbon_cost += leaf_construction_cost_base*leaf_area;
69
70 //internode (cost per volume basis)
71 float internode_volume = 0;
72 uint start_ind = shoot_index.x*phytomer_parameters.internode.length_segments;
73 for( int segment = start_ind; segment<start_ind+phytomer_parameters.internode.length_segments; segment++ ){
74 internode_volume += context_ptr->getTubeObjectSegmentVolume( parent_shoot_ptr->internode_tube_objID, segment );
75 }
76 internode_volume /= current_internode_scale_factor; //convert to fully-elongated volume
77 phytomer_carbon_cost += internode_construction_cost_base*internode_volume;
78
79 return phytomer_carbon_cost;
