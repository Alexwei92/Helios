317  //parallel cyclic reduction
318  for (int j = 0; j <iteration; j++)

321

326  int iLeft = i-delta;

329 
330  assert( i>0 && iLeft>0 && iRight<=systemSize );
331

333  float tmp2 = c[i] / b[iRight];
334 
335  b[i] = b[i] - c[iLeft] * tmp1 - a[iRight] * tmp2;
336  d[i] = d[i] - d[iLeft] * tmp1 - d[iRight] * tmp2;
337  a[i] = -a[iLeft] * tmp1;
338  c[i] = -c[iRight] * tmp2;

343  }

348  float tmp3 = b[addr2]*b[addr1]-c[addr1]*a[addr2];
