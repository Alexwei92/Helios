646 double toc( const char* message ) const {
647 if (!running) {
648 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." << std::endl;
649 return 0;
650 }

652 auto timer_end = std::chrono::high_resolution_clock::now();;
653 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
654 if (strcmp(message, "mute")!=0) {
655 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
