502 double toc( const char* message ) const {
503 if (!running) {
504 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." << std::endl;
505 return 0;
506 }
507
508 auto timer_end = std::chrono::high_resolution_clock::now();;
509 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
510 if (strcmp(message, "mute")!=0) {
511 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
512 }
513 return duration;
514 }
