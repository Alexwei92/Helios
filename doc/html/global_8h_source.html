417 double toc( const char* message ) const {
418 if (!running) {
419 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..."
420 << std::endl;
421 return 0;
422 }
423
424 auto timer_end = std::chrono::high_resolution_clock::now();;
425 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
426 if (strcmp(message, "mute")!=0) {
427 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
428 }
429 return duration;
