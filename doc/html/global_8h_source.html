615 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." << std::endl;
616 return 0;
617 }

620 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
621 if (strcmp(message, "mute")!=0) {
