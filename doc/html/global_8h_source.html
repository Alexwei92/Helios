551 double toc( const char* message ) const {
552 if (!running) {
553 std::cerr << "ERROR (Timer): You must call `tic' before calling `toc'. Ignoring call to `toc'..." << std::endl;
554 return 0;
555 }
556
557 auto timer_end = std::chrono::high_resolution_clock::now();;
558 double duration = std::chrono::duration<double>(timer_end - timer_start).count();
559 if (strcmp(message, "mute")!=0) {
560 std::cout << "Elapsed time is " << duration << " seconds: " << message << std::endl;
561 }
562 return duration;
